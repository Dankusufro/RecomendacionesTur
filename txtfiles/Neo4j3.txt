// PARTE 2: Crear relaciones con categorías para los nuevos lugares
MATCH (l:LugarTuristico), (c:Categoria)
WHERE l.categoria = c.nombre 
AND l.nombre IN ["Templo del Cielo", "Palacio de Verano", "Valle Sagrado", "Ollantaytambo", 
                 "Foro Romano", "Monte Palatino", "Mehtab Bagh", "Tumba de Itimad-ud-Daulah", 
                 "Los Inválidos", "Museo de Orsay"]
CREATE (l)-[:PERTENECE_A]->(c);
