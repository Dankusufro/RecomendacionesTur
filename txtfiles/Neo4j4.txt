// PARTE 2: Crear relaciones con categorías para los nuevos lugares de naturaleza
MATCH (l:LugarTuristico), (c:Categoria)
WHERE l.categoria = c.nombre 
AND l.nombre IN ["Antelope Canyon", "Horseshoe Bend", "High Line Park", "Brooklyn Bridge Park", 
                 "Hakone", "Lago Kawaguchi", "Mykonos", "Lago Titicaca", 
                 "Río Ganges en Varanasi", "Playa de Ipanema"]
CREATE (l)-[:PERTENECE_A]->(c);

// PARTE 3: Crear relaciones CERCA_DE entre lugares de naturaleza

// Conexiones con Gran Cañón
MATCH (grand_canyon:LugarTuristico {nombre: "Gran Cañón"}), 
      (antelope_canyon:LugarTuristico {nombre: "Antelope Canyon"})
CREATE (grand_canyon)-[:CERCA_DE {distancia_km: 240}]->(antelope_canyon),
       (antelope_canyon)-[:CERCA_DE {distancia_km: 240}]->(grand_canyon);

      (horseshoe_bend:LugarTuristico {nombre: "Horseshoe Bend"})

       (horseshoe_bend)-[:CERCA_DE {distancia_km: 220}]->(grand_canyon);
