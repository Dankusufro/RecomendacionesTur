// PARTE 2: Crear relaciones con categorías para los nuevos lugares
MATCH (l:LugarTuristico), (c:Categoria)
WHERE l.categoria = c.nombre 
AND l.nombre IN ["Partenón", "Santorini", "Taj Mahal", "Fuerte Rojo", "Ópera de Sídney", 
                 "Puente del Puerto de Sídney", "Puerta de Brandemburgo", "Castillo de Neuschwanstein", 
                 "Cataratas del Niágara", "Plaza Roja"]
CREATE (l)-[:PERTENECE_A]->(c);

// PARTE 3: Crear relaciones CERCA_DE para los nuevos lugares
